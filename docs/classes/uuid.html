Class UUID

内置的 UUID（通用唯一识别码）服务，版本为 Version 1 (date-time and MAC address)。

import { ToraComponent, Post, UUID } from 'tora'

@ToraComponent()
class TestRouter {

    constructor(
        private uuid: UUID,
    ) {
    }

    generate_id() {
        // 生成新的 id。
        const new_id = this.uuid.create()
        return { id: new_id }
    }
}
